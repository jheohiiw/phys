\textbf{Given:}
q_1 = -2.80\,\mu C = -2.80\times10^{-6}C
\qquad
q_2 = -7.80\,\mu C = -7.80\times10^{-6}C
\qquad
m = 1.50\,g = 1.50\times10^{-3}kg
\qquad
r_1=0.800m,\ v_1=22.0m/s
\qquad
r_2=0.400m,\ v_2=?

\textbf{Step 1: Use conservation of mechanical energy}
\qquad K_1 + U_1 = K_2 + U_2

\textbf{Step 2: Write kinetic energies}
\qquad K_1=\frac12 m v_1^2
\qquad
K_2=\frac12 m v_2^2

\textbf{Step 3: Write electric potential energy}
\qquad U = k\frac{q_1 q_2}{r}

\textbf{Step 4: Substitute into energy equation}
\qquad \frac12 m v_1^2 + k\frac{q_1 q_2}{r_1}
=
\frac12 m v_2^2 + k\frac{q_1 q_2}{r_2}

\textbf{Step 5: Solve for } v_2^2
\qquad \frac12 m v_2^2
=
\frac12 m v_1^2 + k q_1 q_2\left(\frac{1}{r_1}-\frac{1}{r_2}\right)

\qquad v_2^2
=
v_1^2 + \frac{2k q_1 q_2}{m}\left(\frac{1}{r_1}-\frac{1}{r_2}\right)

\textbf{Step 6: Compute pieces}
v_1^2 = (22.0)^2 = 484

\qquad q_1q_2 = (-2.80\times10^{-6})(-7.80\times10^{-6})
= 2.184\times10^{-11}

\qquad \left(\frac{1}{r_1}-\frac{1}{r_2}\right)
=
\left(\frac{1}{0.800}-\frac{1}{0.400}\right)
=
(1.25-2.50)=-1.25

\textbf{Step 7: Plug values}
\qquad v_2^2
=
484
+
\frac{2(9.0\times10^9)(2.184\times10^{-11})}{1.50\times10^{-3}}
(-1.25)

\textbf{Step 8: Simplify numerator}
2(9.0\times10^9)(2.184\times10^{-11}) = 0.39312

\qquad \frac{0.39312}{1.50\times10^{-3}} = 262.08

\qquad 262.08(-1.25)=-327.6

\textbf{Step 9: Final}
\qquad v_2^2 = 484 - 327.6 = 156.4
\qquad v_2 = \sqrt{156.4} \approx 12.5\,m/s

\boxed{v_2 \approx 12.5\ \text{m/s}}
