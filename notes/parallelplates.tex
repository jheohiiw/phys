\textbf{Given:}
\quad \text{Two large parallel plates, separation } d
\quad \text{Electric field halfway has magnitude } E

\textbf{Question:}
\quad \text{If separation becomes } d/2,\ \text{what is the new field halfway?}

\textbf{Step 1: Recall the field between (large) parallel plates}
\qquad E = \frac{\sigma}{\varepsilon_0}

\textbf{Step 2: Identify what changes}
\qquad \text{Only the distance changes: } d \to d/2
\qquad \text{The surface charge density } \sigma \text{ is the same (same plates).}

\textbf{Step 3: Use the formula}
\qquad E_{\text{new}} = \frac{\sigma}{\varepsilon_0}

\textbf{Step 4: Compare with original}
\qquad E_{\text{old}} = \frac{\sigma}{\varepsilon_0}
\qquad \Rightarrow E_{\text{new}} = E_{\text{old}}

\boxed{E_{\text{new}} = E}
