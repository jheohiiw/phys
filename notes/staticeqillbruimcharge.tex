\textbf{Given:}
\quad Q=-3.1\,nC \text{ is in the middle}
\quad \text{Distances: } q_1 \text{ to } Q \text{ is } 10cm,\ Q \text{ to } q_2 \text{ is } 10cm
\quad \Rightarrow \text{distance } q_1 \text{ to } q_2 \text{ is } 20cm

\textbf{Goal:}
\quad \text{Find } q_1 \text{ so that } q_2 \text{ has net force } 0.

\textbf{Step 1: Forces on } q_2
\qquad \text{Force from } Q: \quad F_Q = k\frac{|Qq_2|}{(0.10)^2}
\qquad \text{Force from } q_1: \quad F_{q_1} = k\frac{|q_1 q_2|}{(0.20)^2}

\textbf{Step 2: Equilibrium means magnitudes equal}
\qquad F_Q = F_{q_1}

\textbf{Step 3: Set them equal}
\qquad k\frac{|Qq_2|}{(0.10)^2} = k\frac{|q_1 q_2|}{(0.20)^2}

\textbf{Step 4: Cancel } k \text{ and } |q_2|
\qquad \frac{|Q|}{(0.10)^2} = \frac{|q_1|}{(0.20)^2}

\textbf{Step 5: Solve for } |q_1|
\qquad |q_1| = |Q|\frac{(0.20)^2}{(0.10)^2}
= |Q|\frac{0.04}{0.01}=4|Q|

\textbf{Step 6: Plug } |Q|=3.1\,nC
\qquad |q_1| = 4(3.1\,nC)=12.4\,nC \approx 12\,nC

\boxed{q_1 \approx 12\ nC}
