\textbf{Given:}
q_1=+4.0\mu C,\ q_2=-4.0\mu C

\textbf{Step 1: Electric potential from point charges}
\qquad V = k\left(\frac{q_1}{r_1}+\frac{q_2}{r_2}\right)

\textbf{Step 2: Distances from the diagram}
\qquad \text{At A: } r_{A1}=0.30m,\ r_{A2}=0.50m
\qquad \text{At B: } r_{B1}=0.50m,\ r_{B2}=0.30m

\textbf{Step 3: Compute } V_A
\qquad V_A = 9.0\times10^9\left(\frac{4.0\times10^{-6}}{0.30}+\frac{-4.0\times10^{-6}}{0.50}\right)

\qquad \frac{4.0\times10^{-6}}{0.30}=1.333\ldots\times10^{-5}
\qquad \frac{-4.0\times10^{-6}}{0.50}=-8.0\times10^{-6}

\qquad \text{Sum}=5.333\ldots\times10^{-6}
\qquad V_A = 9.0\times10^9(5.333\ldots\times10^{-6})\approx 4.8\times10^4V=48kV

\textbf{Step 4: Compute } V_B
\qquad V_B = 9.0\times10^9\left(\frac{4.0\times10^{-6}}{0.50}+\frac{-4.0\times10^{-6}}{0.30}\right)

\qquad \frac{4.0\times10^{-6}}{0.50}=8.0\times10^{-6}
\qquad \frac{-4.0\times10^{-6}}{0.30}=-1.333\ldots\times10^{-5}

\qquad \text{Sum}=-5.333\ldots\times10^{-6}
\qquad V_B \approx -48kV

\textbf{Step 5: Difference}
\qquad V_A - V_B = 48kV - (-48kV)=96kV

\boxed{V_A - V_B = 96\ kV}
