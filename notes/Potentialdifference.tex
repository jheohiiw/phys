\section*{Problem 3: Potential Difference $V_A - V_B$}

Two point charges:
\[
q_1 = +4.0\,\mu C, \quad q_2 = -4.0\,\mu C
\]

Distances from geometry:
\[
r_{A1} = 0.30\,m, \quad r_{A2} = 0.50\,m
\]
\[
r_{B1} = 0.50\,m, \quad r_{B2} = 0.30\,m
\]

\subsection*{Step 1: Electric Potential Formula}

\[
V = k\left(\frac{q}{r}\right)
\]

\[
k = 9.0 \times 10^9 \, N\cdot m^2/C^2
\]

\subsection*{Step 2: Potential at A}

\[
V_A =
k\left(
\frac{4.0\times10^{-6}}{0.30}
-
\frac{4.0\times10^{-6}}{0.50}
\right)
\]

\[
V_A =
9.0\times10^9
\left(
1.33\times10^{-5}
-
8.0\times10^{-6}
\right)
\]

\[
V_A = 9.0\times10^9(5.33\times10^{-6})
\]

\[
V_A = 4.8\times10^4 \, V
\]

\[
V_A = 48\,kV
\]

\subsection*{Step 3: Potential at B}

\[
V_B =
k\left(
\frac{4.0\times10^{-6}}{0.50}
-
\frac{4.0\times10^{-6}}{0.30}
\right)
\]

\[
V_B = -48\,kV
\]

\subsection*{Step 4: Potential Difference}

\[
V_A - V_B
=
48\,kV - (-48\,kV)
\]

\[
\boxed{V_A - V_B = 96\,kV}
\]
