\section*{Problem 3 (Speed at 0.400 m)}

Given:
\[
q_1 = -2.80\,\mu C
\]
\[
q_2 = -7.80\,\mu C
\]
\[
m = 1.50\times10^{-3}\,kg
\]
\[
r_1 = 0.800\,m, \quad r_2 = 0.400\,m
\]
\[
v_1 = 22.0\,m/s
\]

\subsection*{Step 1: Energy Conservation}

\[
\frac{1}{2}m(v_2^2 - v_1^2)
=
k q_1 q_2
\left(
\frac{1}{r_1}
-
\frac{1}{r_2}
\right)
\]

\subsection*{Step 2: Compute Constant Term}

\[
k q_1 q_2 =
(9.0\times10^9)
(2.184\times10^{-11})
=
0.19656
\]

\subsection*{Step 3: Distance Term}

\[
\frac{1}{0.800}
-
\frac{1}{0.400}
=
-1.25
\]

\[
\Delta U = -0.2457\,J
\]

\subsection*{Step 4: Solve for $v_2$}

\[
0.00075(v_2^2 - 484)
=
0.2457
\]

\[
v_2^2 = 811.6
\]

\[
\boxed{v_2 = 28.5\,m/s}
\]
